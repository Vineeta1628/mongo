📐 Database Schema Design

Collection: users

User Schema Structure 
{
  _id: ObjectId,
  email: String (unique, indexed),
  passwordHash: String,
  profile: {
    firstName: String,
    lastName: String,
    phone: String
  },
  addresses: [{
    street: String,
    city: String,
    state: String,
    zipCode: String,
    isDefault: Boolean
  }],
  createdAt: Date,
  lastLogin: Date
}

Collection: products

Product Schema Structure 
{
  _id: ObjectId,
  name: String (text indexed),
  description: String (text indexed),
  category: String (indexed),
  price: Number,
  stock: Number,
  images: [String],
  specifications: Object,
  rating: {
    average: Number,
    count: Number
  },
  tags: [String]
}

Collection: orders

Order Schema Structure 
{
  _id: ObjectId,
  orderNumber: String (unique),
  userId: ObjectId (reference),
  items: [{
    productId: ObjectId,
    productName: String,
    quantity: Number,
    price: Number,
    subtotal: Number
  }],
  totalAmount: Number,
  status: String,
  shippingAddress: Object,
  payment: {
    method: String,
    status: String,
    transactionId: String
  },
  createdAt: Date,
  deliveredAt: Date
}

// Users indexes
db.users.createIndex({ email: 1 }, { unique: true })

// Products indexes
db.products.createIndex({ name: "text", description: "text" })
db.products.createIndex({ category: 1, price: 1 })

// Orders indexes
db.orders.createIndex({ orderNumber: 1 }, { unique: true })
db.orders.createIndex({ userId: 1, createdAt: -1 })
db.orders.createIndex({ status: 1 })

🛒 Phase 3: Shopping Cart

Add to Cart

var user = db.users.findOne({ email: "john.doe@email.com" })
var product = db.products.findOne({ name: "MacBook Pro 16-inch" })

db.carts.updateOne(
  { userId: user._id },
  {
    $push: {
      items: {
        productId: product._id,
        productName: product.name,
        quantity: 1,
        price: product.price,
        subtotal: product.price
      }
    },
    $inc: { totalAmount: product.price },
    $set: { updatedAt: new Date() }
  },
  { upsert: true }
)

View Cart with Product Details

db.carts.aggregate([
  { $match: { userId: user._id } },
  { $unwind: "$items" },
  {
    $lookup: {
      from: "products",
      localField: "items.productId",
      foreignField: "_id",
      as: "productDetails"
    }
  },
  { $unwind: "$productDetails" },
  {
    $group: {
      _id: "$_id",
      items: {
        $push: {
          productName: "$productDetails.name",
          quantity: "$items.quantity",
          price: "$items.price",
          subtotal: "$items.subtotal"
        }
      },
      totalAmount: { $first: "$totalAmount" }
    }
  }
])

📋 Phase 4: Order Processing

Create Order from Cart

var cart = db.carts.findOne({ userId: user._id })
var orderNumber = "ORD-" + new Date().getTime()
var defaultAddress = user.addresses.find(addr => addr.isDefault)

db.orders.insertOne({
  orderNumber: orderNumber,
  userId: user._id,
  items: cart.items,
  totalAmount: cart.totalAmount,
  status: "pending",
  shippingAddress: defaultAddress,
  payment: {
    method: "credit_card",
    status: "pending"
  },
  createdAt: new Date()
})

// Clear cart
db.carts.updateOne(
  { userId: user._id },
  { $set: { items: [], totalAmount: 0 } }
)

Update Order Status

db.orders.updateOne(
  { orderNumber: orderNumber },
  {
    $set: {
      status: "delivered",
      deliveredAt: new Date()
    }
  }
)

⭐ Phase 5: Reviews & Ratings

Add Product Review

db.reviews.insertOne({
  productId: product._id,
  userId: user._id,
  rating: 5,
  title: "Best laptop ever!",
  comment: "Amazing performance and battery life",
  helpful: 0,
  createdAt: new Date()
})

Update Product Rating

var stats = db.reviews.aggregate([
  { $match: { productId: product._id } },
  {
    $group: {
      _id: null,
      averageRating: { $avg: "$rating" },
      totalReviews: { $sum: 1 }
    }
  }
]).toArray()[0]

db.products.updateOne(
  { _id: product._id },
  {
    $set: {
      "rating.average": Math.round(stats.averageRating * 10) / 10,
      "rating.count": stats.totalReviews
    }
  }
)

📊 Phase 6: Analytics & Reporting

Total Revenue

db.orders.aggregate([
  { $match: { status: "delivered" } },
  {
    $group: {
      _id: null,
      totalRevenue: { $sum: "$totalAmount" },
      totalOrders: { $sum: 1 },
      averageOrderValue: { $avg: "$totalAmount" }
    }
  }
])

Top Selling Products

db.orders.aggregate([
  { $match: { status: { $ne: "cancelled" } } },
  { $unwind: "$items" },
  {
    $group: {
      _id: "$items.productId",
      productName: { $first: "$items.productName" },
      totalSold: { $sum: "$items.quantity" },
      revenue: { $sum: "$items.subtotal" }
    }
  },
  { $sort: { totalSold: -1 } },
  { $limit: 10 }
])

Customer Lifetime Value

db.orders.aggregate([
  { $match: { status: { $ne: "cancelled" } } },
  {
    $group: {
      _id: "$userId",
      totalSpent: { $sum: "$totalAmount" },
      orderCount: { $sum: 1 }
    }
  },
  { $sort: { totalSpent: -1 } },
  { $limit: 10 }
])

⚡ Performance Optimization

💡 Optimization Tips:
  • Use compound indexes for common query patterns
  • Put $match stages early in aggregation pipelines
  • Use projection to limit returned fields
  • Monitor slow queries with explain()
  • Implement proper connection pooling
